```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: ingress-wear-watch
  namespace: app-space
spec:
  rules:
  - http:
      paths:
      - path: /wear
        pathType: ImplementationSpecific
        backend:
          service:
            name: wear-service
            port:
              number: 8080
      - path: /stream
        pathType: ImplementationSpecific
        backend:
          service:
            name: video-service
            port:
              number: 8080
```
